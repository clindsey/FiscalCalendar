Handlebars.registerHelper "yearTitle", (quarters) ->
  years = []

  for quarter in quarters
    for month in quarter.months
      years = years.concat _.pluck month.weeks, 'year'

  +_.chain(years).countBy().pairs().max(_.last).head()
